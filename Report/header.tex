% Load required packages
\usepackage{fancyhdr}
\usepackage{graphicx}

% Configure fancy headers and footers
\pagestyle{fancy}

% Clear default headers and footers
\fancyhf{}

% Remove the header rule line
\renewcommand{\headrulewidth}{0pt}

% Place IPL logo on top-right of every page
\fancyhead[R]{\includegraphics[height=0.8cm]{ipl_logo.png}}

% Add centered page numbers in the footer
\fancyfoot[C]{\thepage}

% Apply fancy style to plain pages (like first page)
\fancypagestyle{plain}{
  \fancyhf{}
  \renewcommand{\headrulewidth}{0pt}
  \fancyhead[R]{\includegraphics[height=0.8cm]{ipl_logo.png}}
  \fancyfoot[C]{\thepage}
}
